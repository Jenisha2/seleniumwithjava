CREATE TABLE empdetails (
  DeptID VARCHAR(50),
  EmpName VARCHAR(50),
  Salary int
);

ANSWER1:

select * from (
  select *, DENSE_RANK() OVER(PARTITION by deptid ORDER by salary DESC) as r
  from empdetails
)
where r = 1;

ANSWER2:

select * from empdetails where (deptid, salary) in 
(
  select deptid, MAX(salary)
from empdetails
GROUP By deptid)
